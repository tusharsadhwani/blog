from functools import wraps
def bang(func):
    @wraps(func)
    def inner(*args, **kwargs):
        string = func(*args, **kwargs)
        return string + '!'

    return inner

@bang
def greet(name):
    return f'Hello, {name}'

print(greet('Tushar'))
